import React from 'react';
import checkPropTypes from 'check-prop-types';
import { shallow } from 'enzyme';
import { Userdata } from './Userdata';
import { formatPhpTimestamp } from '../../../../utils/utils';

describe('Header > Userdata', () => {
  const baseProps = {
    fullname: 'dr. Lantos Tibor',
    defaultStartTime: 480,
    firstDay: 1556696327,
    lastDay: 1559288327,
    office: 'KIFÜ - Victor Hugo',
    taxNumber: '8449060281',
  };
  const component = shallow(<Userdata {...baseProps} />);

  test('renders without errors', () => {
    expect(component.length).toEqual(1);
  });

  test('renders without propType warning', () => {
    const propError = checkPropTypes(
      Userdata.propTypes,
      baseProps,
      'prop',
      Userdata.name,
    );
    expect(propError).toBeUndefined();
  });

  test('renders sub-component LeftDisplay without errors', () => {
    expect(component.find('LeftDisplay').length).toEqual(1);
  });

  test('renders sub-component RightDisplay without errors', () => {
    expect(component.find('RightDisplay').length).toEqual(1);
  });
});
